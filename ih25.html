            abstract class Bird {
                abstract void fly();
                abstract void makeSound();
            }
            
            class Eagle extends Bird {
                void fly() {
                    System.out.println("The eagle flies very high.");
                }
            
                 void makeSound() {
                    System.out.println("The eagle screeches loudly.");
                }
            }
            
            class Hawk extends Bird {
                void fly() {
                    System.out.println("The hawk glides swiftly through the sky.");
                }
            
                    public void makeSound() {
                    System.out.println("The hawk makes a sharp call.");
                }
            }
            
            
            class Main {
                public static void main(String[] args) {
                    Bird obj = new Eagle();
                    Bird obj2 = new Hawk();
            
                     obj.fly();
                    obj.makeSound();
            
                    obj2.fly();
                    obj2.makeSound();
                }
            }
