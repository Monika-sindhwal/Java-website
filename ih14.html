            abstract class Instrument {
                abstract void play();
                abstract void tune();
            }
            
            class Glockenspiel extends Instrument {
              
                void play() {
                    System.out.println("Playing the Glockenspiel.");
                }
            
                void tune() {
                    System.out.println("Tuning the Glockenspiel.");
                }
            }
            
            class Violin extends Instrument {
             
                void play() {
                    System.out.println("Playing the Violin .");
                }
            
                 void tune() {
                    System.out.println("Tuning the Violin.");
                }
            }
            class Main {
                public static void main(String[] args) {
                    Instrument obj = new Glockenspiel();
                    Instrument obj2 = new Violin();
            
                    obj.play();
                    obj.tune();
            
                    obj2.play();
                    obj2.tune();
                }
            }
