            class A {

                    double b = scanner.nextDouble();
                    
                    System.out.print("Enter the exponent: ");
                    double e = scanner.nextDouble();
                    
                    double result = Math.pow(b, e);
                    
                    System.out.println(b + " raised to the power of " + e + " is: " + result);
                }
            }
