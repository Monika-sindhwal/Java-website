          class TrafficLight {
            String color;
            int duration;
        
            public TrafficLight(String c, int d) {
                color = c;
                duration = d;
            }
        
            public void changeColor(String newColor, int newDuration) {
                color = newColor;
                duration = newDuration;
            }
        
            public boolean isRed() {
                return this.color.equalsIgnoreCase("red");
            }
        
            public boolean isGreen() {
                return this.color.equalsIgnoreCase("green");
            }
        
            public String getColor() {
                return color;
            }
        
            public int getDuration() {
                return duration;
            }
        }
        
        class Main {
            public static void main(String[] args) {
                TrafficLight light = new TrafficLight("Red", 30);
                System.out.println("Current color: " + light.getColor());
                System.out.println("Is it red? " + light.isRed());
                
                light.changeColor("Green", 45);
                System.out.println("Current color: " + light.getColor());
                System.out.println("Is it green? " + light.isGreen());
            }
        }
