                abstract double calculateVolume(int r);
                abstract double calculateSurfaceArea(int r);
                }
                class Sphere extends Shape3D{
                double calculateVolume(int r){
                return ((4/3)*3.14*r*r*r);
                
                }
                double calculateSurfaceArea(int r){
                return (4*3.14*r*r);
                }
                
                }

                double calculateVolume(int r){
                return (r*r*r);

                }
