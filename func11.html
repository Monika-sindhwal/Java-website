                   public static boolean consecutivechecker(int a, int b, int c) {
                    int min = Math.min(a, Math.min(b, c));
                    int max = Math.max(a, Math.max(b, c));
                    int middle = a + b + c - min - max;
            
                    return (max - middle == 1) && (middle - min == 1);
                }
            }
