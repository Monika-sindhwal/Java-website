            import java.io.*;

class A {
    public static void main(String[] args) throws IOException {
       
 BufferedReader reader = new BufferedReader(new     InputStreamReader(System.in));
            
             System.out.print("Enter the size of the array: ");
            int size = Integer.parseInt(reader.readLine());

           
            int[] arr = new int[size];

            
            
            for (int i = 0; i < size; i++) {
System.out.println("Enter " + (i+1) + " element:");
                arr[i] = Integer.parseInt(reader.readLine());
            }

            System.out.print("Enter the element to find the index of: ");
            int find = Integer.parseInt(reader.readLine());

            int ind = -1;

             for (int i = 0; i < arr.length; i++) {
                if (arr[i] == find) {
                    ind = i; 
                    break;                 }
            }

            if (ind != -1) {
                System.out.println("Element " + find + " found at index: " + ind);
            } else {
                System.out.println("Element " + find + " not found in the array.");
            }

             
    }
}
