            import java.io.*;
class A{
static String str;
public static void main(String args[]) throws java.io.IOException{
InputStreamReader obj=new InputStreamReader(System.in);
BufferedReader a = new BufferedReader(obj);
System.out.println("Enter the string:");
str =a.readLine();
int letterCount = 0;
        int spaceCount = 0;
        int numberCount = 0;
        int otherCount = 0;
char c[]=str.toCharArray();

        for (int i=0;i<c.length;i++) {
        
            if (Character.isLetter(c[i])) {
                letterCount++;
            } else if (Character.isSpaceChar(c[i])) {
                spaceCount++;
            } else if (Character.isDigit(c[i])) {
                numberCount++;
            } else {
                otherCount++;
            }
        }
        System.out.println("letter:"+ letterCount + "\nspaces" + spaceCount + "\n number" +numberCount + "\n Other characters" +otherCount);
}
}
