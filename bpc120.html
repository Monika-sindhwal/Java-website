            class A {
                public static void main(String[] args) {
                    double num1 = 100;
                    double num2 = 200;
                    double num3 = 300;
            
                    double average = computeAverage(num1, num2, num3);
                    System.out.println("Average: " + average);
                }
            
                public static double computeAverage(double a, double b, double c) {
                    return (a + b + c) / 3.0;
                }
            }
