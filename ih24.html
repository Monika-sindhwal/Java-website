            abstract class Shape2D {
                abstract void draw();   
                abstract void resize();
            }
            
            class Rectangle extends Shape2D {
                int length;
                int breadth;
            
                Rectangle(int l, int b) {
                    length = l;
                    breadth = b;
                }
            
                void draw() {
                    System.out.println("Drawing a rectangle of length=" + length + " and breadth=" + breadth);
                }
            
                void resize() {
                    length += 1;
                    breadth += 1;
                    System.out.println("Resizing the rectangle to length=" + length + " and breadth=" + breadth);
                }
            }
            
            class Circle extends Shape2D {
                int radius;
            
               Circle(int r) {
                    radius = r;
                }
            
                void draw() {
                    System.out.println("Drawing a circle of radius=" + radius);
                }
            
                void resize() {
             
                    radius += 1;
                    System.out.println("Resizing the circle to radius=" + radius);
                }
            }
            
            class Main {
                public static void main(String[] args) {
                    Circle obj = new Circle(7);
                    obj.draw();      
                    obj.resize();    
            
                    Rectangle obj2 = new Rectangle(20, 40);
                    obj2.draw();            
                    obj2.resize(); 
                 }
            }
