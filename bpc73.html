            import java.util.Scanner;

            class A {
                public static void main(String[] args) {
                   
                    Scanner scanner = new Scanner(System.in);
            
                    System.out.print("Enter a string: ");
                    String s1 = scanner.nextLine();
            
                    System.out.print("Enter the starting position (0-based index): ");
                    int start = scanner.nextInt();
                    System.out.print("Enter the ending position (0-based index): ");
                    int end = scanner.nextInt();
            
                   
                    if (start < 0 || end > s1.length() || start >= end) {
                        System.out.println("Invalid positions.");
                    } else {
                        
                        String substring = s1.substring(start, end);
                       
                        System.out.println("The substring between positions " + start + " and " + end + " is: \"" + substring + "\"");
                    }
            
                   
                    scanner.close();
                }
            }
