                    int[] arr = {100, 15, 27, 31, 40, 50, 60};
            
                    System.out.println("Even numbers in the array are:");
                    for (int i = 0; i < arr.length; i++) {
                        if (arr[i] % 2 == 0) { 
                            System.out.print(arr[i] + " ");
                        }
                    }
            
                    // Print odd numbers
                    System.out.println("\nOdd numbers in the array are:");
                    for (int i = 0; i < arr.length; i++) {
                        if (arr[i] % 2 != 0) { 
                            System.out.print(arr[i] + " ");
                        }
                    }
                }
            }
