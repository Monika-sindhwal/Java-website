            class Main {
                public static void main(String[] args) {
                    String initialString = "Hello World";
                    System.out.println("Initial String: " + initialString);
            
                    char[] charactersToAdd = {'!', 'W', 'e', 'l', 'c', 'o', 'm','e'};
            
                    StringBuilder stringBuilder = new StringBuilder(initialString);
            
                    for (int i = 0; i < charactersToAdd.length; i++) {
                        stringBuilder.append(charactersToAdd[i]); 
                    }
            
                    String finalString = stringBuilder.toString();
                    System.out.println("Final String after adding characters: " + finalString);
                }
            }
