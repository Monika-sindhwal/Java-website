            class myThread extends Thread {
                public void run() {
                    for (int i = 0; i < 5; i++) {
                        System.out.println("Name:" + this.getName() + " Priority:" + this.getPriority());
            
                    }
            
                }
            }
            
            class Main {
                public static void main(String x[]) {
                    myThread obj = new myThread();
                    obj.setName("Thread1");
                    obj.setPriority(10);
                    myThread obj2 = new myThread();
                    obj2.setName("Thread2");
                    obj2.setPriority(2);
                    obj.start();
                    obj2.start();
                }
            }
