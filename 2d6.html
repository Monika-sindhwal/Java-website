            import java.io.*;

            class A{
                   public static void main(String[] args) throws IOException {
                    InputStreamReader isr = new InputStreamReader(System.in);
                    BufferedReader obj = new BufferedReader(isr);            
                    
                    System.out.print("Enter the number of rows (square matrix): ");
                    int rows = Integer.parseInt(obj.readLine());
                    
                    int[][] matrix = new int[rows][rows];
            
                    System.out.println("Enter the matrix elements:");
                    for (int i = 0; i < rows; i++) {
                        for (int j = 0; j < rows; j++) {
                            System.out.print("Element [" + i + "][" + j + "]: ");
                            matrix[i][j] = Integer.parseInt(obj.readLine());
                        }
                    }
                    
                    System.out.println("The lower triangular matrix is:");
                    for (int i = 0; i < rows; i++) {
                        for (int j = 0; j < rows; j++) {
                            if (j <= i) {
                                System.out.print(matrix[i][j] + "\t");
                            } else {
                                System.out.print("0\t");
                            }
                        }
                        System.out.println(); 
                    }
                }
            }
