            class A {
                public static void main(String[] args) {
                    int num1 = 15;
                    int num2 = 8;
                    int num3 = 22;
            
                    int smallest = find(num1, num2, num3);
                    System.out.println("Smallest number: " + smallest);
                }
            
                public static int find(int a, int b, int c) {
                    int smallest = a;
                    if (b < smallest) {
                        smallest = b;
                    }
                    if (c < smallest) {
                        smallest = c;
                    }
                    return smallest;
                }
            }
