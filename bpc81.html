            class A {
                public static void main(String[] args) {
                   
                    String s1 = "This is line one.\nThis is line two.\nThis is line three.\nThis is line four.\nhello world";
            
            
                    int count = countLines(s1);
            
                  
                    System.out.println("Total number of lines: " + count);
                }
            
               
                public static int countLines(String str) {
                    
                    str = str.trim();
                    
                   
                    if (str.isEmpty()) {
                        return 0;
                    }
            
                 
                    String[] lines = str.split("\n");
                    return lines.length;
                }
            }
