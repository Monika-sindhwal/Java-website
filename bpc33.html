            import java.util.Scanner;

            class A {
                public static void main(String[] args) {
                    Scanner scanner = new Scanner(System.in);
            
                    System.out.println("Choose the type of prism:");
                    System.out.println("1. Rectangular Prism");
                    System.out.println("2. Triangular Prism");
                    System.out.print("Enter your choice (1 or 2): ");
                    int choice = scanner.nextInt();
            
                    switch (choice) {
                        case 1:
                         
                            System.out.print("Enter the length of the prism: ");
                            double length = scanner.nextDouble();
                            System.out.print("Enter the width of the prism: ");
                            double width = scanner.nextDouble();
                            System.out.print("Enter the height of the prism: ");
                            double height = scanner.nextDouble();
            
                            double rectangularArea = 2 * (length * width + length * height + width * height);
                            System.out.println("Surface area of the Rectangular Prism: " + rectangularArea);
                            break;
            
                        case 2:
                            
                            System.out.print("Enter the base of the triangular base: ");
                            double base = scanner.nextDouble();
                            System.out.print("Enter the height of the triangular base: ");
                            double triangleHeight = scanner.nextDouble();
                            System.out.print("Enter the height of the prism: ");
                            double prismHeight = scanner.nextDouble();
            
                            double triangularBaseArea = 0.5 * base * triangleHeight;
                            double triangularArea = (base * prismHeight) + 2 * triangularBaseArea;
                            System.out.println("Surface area of the Triangular Prism: " + triangularArea);
                            break;
            
                        default:
                            System.out.println("Invalid choice. Please choose either 1 or 2.");
                    }
            
                    
                    scanner.close();
                }
            }
