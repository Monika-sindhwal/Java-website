            import java.util.Scanner;

            class A {
            
                
                public static boolean haveSameCharacters(String str1, String str2) {
                   
                    char[] charArray1 = str1.toCharArray();
                    char[] charArray2 = str2.toCharArray();
                    
                    java.util.Arrays.sort(charArray1);
                    java.util.Arrays.sort(charArray2);
            
                    return java.util.Arrays.equals(charArray1, charArray2);
                }
            
                public static void main(String[] args) {
                    Scanner scanner = new Scanner(System.in);
            
                    System.out.print("Enter the first string: ");
                    String str1 = scanner.nextLine();
            
                    System.out.print("Enter the second string: ");
                    String str2 = scanner.nextLine();
            
                    if (haveSameCharacters(str1, str2)) {
                        System.out.println("The strings are created with the same characters.");
                    } else {
                        System.out.println("The strings are NOT created with the same characters.");
                    }
            
                    scanner.close();
                }
            }
