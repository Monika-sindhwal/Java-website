            import java.util.Scanner;

            class A {
                public static double calculatePentagonArea(double sideLength) {
                    return (1.0 / 4) * Math.sqrt(5 * (5 + 2 * Math.sqrt(5))) * Math.pow(sideLength, 2);
                }
            
                public static void main(String[] args) {
                    Scanner scanner = new Scanner(System.in);
                    
                    System.out.print("Enter the length of a side of the pentagon: ");
                    double sideLength = scanner.nextDouble();
                    
                    double area = calculatePentagonArea(sideLength);
                    System.out.printf("The area of the pentagon is: %.2f", area);
                }
            }
