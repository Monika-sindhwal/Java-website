            class Library{
                String books[];
                int capacity;
                int bookCount;
                Library(int cap){
                books=new String [cap];
                int bookCount=0;
                }
                void bookAdd(String book){
                if(bookCount>=books.length){
                System.out.println("Cannot add more books");
                return;}
                for(int i=0;i < bookCount;i++){
                if(book.equals(books[i])){
                System.out.println(book+" is already present in library");
                return;}
                }
                books[bookCount]=book;
                bookCount++;
                System.out.println(book + " added to the library.");
                    }
                 public void removeBook(String book) {
                        for (int i = 0; i < bookCount; i++) {
                            if (books[i].equals(book)) {
                                for (int j = i; j < bookCount - 1; j++) {
                                    books[j] = books[j + 1];
                                }
                                books[bookCount - 1] = null; 
                                bookCount--;                                    System.out.println(book + " removed from the library.");
                                return;
                            }
                  }
                        System.out.println(book + " is not in the library.");
                    }
                
                void display(){
                if(bookCount==0){
                System.out.println("No book present.");
                        } else {
                            for (int i = 0; i < bookCount; i++) {
                                System.out.println((i + 1) + ". " + books[i]);
                            }
                        }
                    }
                public static void main(String x[]){
                Library obj=new Library(20);
