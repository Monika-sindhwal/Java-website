            class A {
                public static void main(String[] args) {
                    int[] array = {5, 17, 8, 10, 4, 212};
            
                    int even = 0;
                    int odd = 0;
            
                    for (int i = 0; i < array.length; i++) {
                        if (i % 2 == 0)
                            even++;
                        else
                            odd++;
                    }
            
                    System.out.println("even integers in array are: " + even);
                    System.out.println("odd integers in array are: " + odd);
                }
            }
