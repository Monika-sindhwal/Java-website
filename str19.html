          class A {
            public static void main(String[] args) {
                String input = "abcHHKKLKKGFDVBj";
                int n = 5;
        
                divideString(input, n);
            }
        
            public static void divideString(String str, int n) {
                int length = str.length();
                if (length % n != 0) {
                    System.out.println("Cannot divide the string into equal parts.");
                    return;
                }
        
                int partLength = length / n;
                for (int i = 0; i < n; i++) {
                    String part = str.substring(i * partLength, (i + 1) * partLength);
                    System.out.println("Part " + (i + 1) + ": " + part);
                }
            }
        }
