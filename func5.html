            class A{

   
                public static double FV(double principal, double annualInterestRate, int years) {
                   
                    double rate = annualInterestRate / 100.0;
            
                    
                    double futureValue = principal * Math.pow((1 + rate), years);
            
                    return futureValue;
                }
            
                public static void main(String[] args) {
                  
                    double principal = 10000.0;
                    double Rate = 5.0; 
                    int years = 20;
            
                    double futureValue = FV(principal, Rate, years);
                    System.out.printf("Future investment value: %.2f%n", futureValue);
                }
            }
