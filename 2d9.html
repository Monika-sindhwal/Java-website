            import java.io.*;

            class A{
                   public static void main(String[] args) throws IOException {
                    InputStreamReader isr = new InputStreamReader(System.in);
                    BufferedReader obj = new BufferedReader(isr);            
                    
                    System.out.print("Enter the number of rows: ");
                    int rows = Integer.parseInt(obj.readLine());
                    System.out.print("Enter the number of columns: ");
                    int columns = Integer.parseInt(obj.readLine());
                    
                    int[][] matrix = new int[rows][columns];
            
                    System.out.println("Enter the matrix elements:");
                    for (int i = 0; i < rows; i++) {
                        for (int j = 0; j < columns; j++) {
                            System.out.print("Element [" + i + "][" + j + "]: ");
                            matrix[i][j] = Integer.parseInt(obj.readLine());
                        }
                    }
            
                    System.out.println("Sum of each row:");
                    for (int i = 0; i < rows; i++) {
                        int rS = 0;
                        for (int j = 0; j < columns; j++) {
                            rS += matrix[i][j];
                        }
                        System.out.println("Sum of row " + i + ": " + rS);
                    }
            
                    System.out.println("Sum of each column:");
                    for (int j = 0; j < columns; j++) {
                        int cS = 0;
                        for (int i = 0; i < rows; i++) {
                            cS += matrix[i][j];
                        }
                        System.out.println("Sum of column " + j + ": " + cS);
                    }
                }
            }
